# jvm parameters
-Dcom.sun.management.jmxremote.port=10001 -Dcom.sun.management.jmxremote.authenticate=true -Dcom.sun.management.jmxremote.ssl=false
# authenticate setting
cd jre/lib/management
# vi jmxremote.access

controlRole   readwrite \
            create javax.management.monitor.*,javax.management.timer.* \
            unregister
# cp jmxremote.password.template jmxremote.password

controlRole password
# change file mode
chmod 600 jmxremote.access
chmod 600 jmxremote.password
# set hosts if `hostname -i` is '127.0.0.1'
real_ip localhost

jconsole => remoteIp:10001
